def diff_vote(inputList, weight=[2, 2, 0.5]):
    yes = 0
    no = 0
    abstain = 0
    for vote in inputList:
        if vote == 'Yes':
            yes += 1
        elif vote == 'No':
            no += 1
        elif vote == 'Abstain':
            abstain += 1
    if (yes**2 + no**2 + abstain**2) > 0:
        diff = 1 - (weight[0] * 2 * (yes * no) + weight[2] * 2 * (no * abstain) + weight[1] * 2 * (yes * abstain)) / (yes + no + abstain)**2
    else:
        diff = np.nan
    return diff
